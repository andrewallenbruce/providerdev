17.3 All RHCs in Georgia

x <- careMain("enrollees")

req <- x@identifier |> 
  request() |> 
  req_url_query(
    "filter[STATE_CD]" = "GA",
    "filter[PROVIDER_TYPE_CD]" = "00-17"
  )

nres <- req |> 
  req_url_path_append("stats") |>
  providertwo:::perform_simple() |> 
  _[["data"]] |> 
  _[["found_rows"]]
