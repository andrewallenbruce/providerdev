                      name,

    state = state,
    name = name,
    number = number

greater_than <- \(arg, value) {
  glue::glue(
  "
  [condition][path]={arg}&
  [condition][operator]=>&
  [condition][value]={value}&
  "
  )
}

greater_than("overall_rating", 3)
[condition][path]=overall_rating&
[condition][operator]=>&
[condition][value]=3&
greater_or_equal <- \(arg, value) {
  glue::glue(
  "filter[fID-1][condition][path]={arg}&\\
  filter[fID-1][condition][operator]=>=&\\
  filter[fID-1][condition][value]={value}"
  )
}

greater_or_equal("overall_rating", 3)
filter[fID-1][condition][path]=overall_rating&filter[fID-1][condition][operator]=>=&filter[fID-1][condition][value]=3

9.2 Query Generation

format_syntax <- \(name, value) {
  setNames(value, paste0(name, "[id-", seq_along(value), "][condition][value]"))
}

format_syntax(name = "filter", value = c("GA", "NY"))
filter[id-1][condition][value] filter[id-2][condition][value]
