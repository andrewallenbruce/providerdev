8.1 Catalog

main_data_raw <- \() {

  x <- RcppSimdJson::fload("https://data.cms.gov/data.json")

  list(
    context             = x[["@context"]],
    id                  = x[["@id"]],
    type                = x[["@type"]],
    conformsTo          = x[["conformsTo"]],
    describedBy         = x[["describedBy"]],
    dataset             = collapse::qTBL(x[["dataset"]]) |>
    collapse::fmutate(
      bureauCode   = delist(bureauCode),
      language     = delist(language),
      programCode  = delist(programCode),
      references   = delist(references),
      theme        = flatten_column(theme),
      keyword      = flatten_column(keyword)) |>
    collapse::frename(remove_at_symbol)
  )
}

select_dataset <- \(x, set) {
  list(
    context             = x[["context"]],
    id                  = x[["id"]],
    type                = x[["type"]],
    conformsTo          = x[["conformsTo"]],
    describedBy         = x[["describedBy"]],
    dataset             = collapse::fsubset(
      x[["dataset"]], 
      fuimus::sf_detect(title, set))
  )
}

query <- glue::glue(
  '
  "filter[fID{fID}][path]" = "{PATH}",
  "filter[fID{fID}][operator]" = "{OPERATOR}",
  "filter[fID{fID}][value]" = "{VALUE}"
  ', 
  fID = 1:2, 
  PATH = c("STATE_CD", "LAST_NAME"), 
  OPERATOR = "=",
  VALUE = c("GA", "SMITH")
  ) |> 
  glue::glue_collapse(sep = ",\n")

query <- glue::glue('c({query})') |> 
  rlang::parse_expr() |> 
  rlang::eval_bare()

request(urls$base) |> 
  req_url_path_append(urls$identifier) |> 
  req_url_path_append("stats") |> 
  req_url_query(!!!query) |> 
  req_perform() |> 
  resp_body_json(simplifyVector = TRUE)

format_syntax <- \(name, value) {
  setNames(
    value, 
    paste0(
      name, 
      "[id-", 
      seq_along(value), 
      "][condition][value]"))
}

format_syntax("filter", c("GA", "NY"))
